        function renderCart() {
            let tQty=0, sPrice=0, html='';
            cart.forEach(i => { 
                tQty+=i.qty; sPrice+=(i.qty*i.price); 
                let cartIcon = i.iconClass.startsWith('http') ? `<img src="${i.iconClass}" class="w-full h-full object-cover rounded">` : `<i class="ph ${i.iconClass} text-xl"></i>`;
                html += `<div class="flex gap-3 items-center border-b pb-3"><div class="w-12 h-12 bg-gray-50 flex items-center justify-center text-gray-400 rounded">${cartIcon}</div><div class="flex-grow"><p class="text-sm font-medium line-clamp-1">${i.name}</p><p class="text-brandRed font-bold text-sm">Rp ${i.price.toLocaleString('id-ID')}</p></div><div class="flex items-center border rounded"><button onclick="updateCartQty('${i.id}', -1)" class="w-6 h-6 bg-gray-50">-</button><span class="text-xs w-6 text-center">${i.qty}</span><button onclick="updateCartQty('${i.id}', 1)" class="w-6 h-6 bg-gray-50">+</button></div><button onclick="removeCartItem('${i.id}')" class="text-gray-400 ml-2"><i class="ph ph-trash"></i></button></div>`; 
            });
            const ongkir = cart.length>0 ? storeConfig.ongkir : 0;
            document.getElementById('cart-list').innerHTML=html; document.getElementById('cart-counter').textContent=tQty; document.getElementById('cart-subtotal').textContent="Rp "+sPrice.toLocaleString('id-ID'); document.getElementById('cart-ongkir-display').textContent="Rp "+ongkir.toLocaleString('id-ID'); document.getElementById('cart-total').textContent="Rp "+(sPrice+ongkir).toLocaleString('id-ID');
            document.getElementById('empty-cart-msg').className = cart.length>0 ? 'hidden' : 'text-center text-gray-400 mt-10';
        }

        async function processCheckout() {
            if(cart.length===0) return showNotification("Kosong", "Keranjang Anda kosong.", "warning");
            if(!currentUser) return openLoginModal();
            const btn = document.getElementById('btn-checkout'); const ot = btn.innerHTML; btn.innerHTML='Memproses...'; btn.disabled=true;
            const sub = cart.reduce((s,i)=>s+(i.price*i.qty),0); const tot = sub + storeConfig.ongkir; const ordId = "ORD-"+Math.floor(1000+Math.random()*9000);
            
            // Format rincian barang untuk disimpan ke database
            let orderItemsData = cart.map(i => ({ name: i.name, price: i.price, qty: i.qty }));
            if (storeConfig.ongkir > 0) orderItemsData.push({ name: "Ongkos Kirim", price: storeConfig.ongkir, qty: 1 });
            const itemsJson = JSON.stringify(orderItemsData);

            let waMsg = `Halo Admin *${storeConfig.name}*, saya konfirmasi pesanan:\n\n👤 *Nama:* ${currentUser.name}\n📦 *Pesanan:*\n`; cart.forEach((i,x) => waMsg+=`${x+1}. ${i.name} (${i.qty}x)\n`); waMsg+=`\n💰 *Total:* Rp ${tot.toLocaleString('id-ID')}\nTerima kasih.`; const waUrl = `https://wa.me/${storeConfig.waNumber}?text=${encodeURIComponent(waMsg)}`;

            }
            btn.innerHTML=ot; btn.disabled=false;
        }

        // Fungsi helper konversi link Google Drive ke Direct Image Link
        function convertDriveLink(url) {
            if (!url) return "";
            const driveRegex = /drive\.google\.com\/file\/d\/([^\/]+)/;
            const match = url.match(driveRegex);
            if (match && match[1]) { return `https://drive.google.com/uc?export=view&id=${match[1]}`; }
            return url;
        }

            document.getElementById('detail-order-id').textContent = o.id;
            document.getElementById('detail-order-user').textContent = o.username;
            document.getElementById('detail-order-total').textContent = "Rp " + Number(o.total).toLocaleString('id-ID');
            
            const list = document.getElementById('detail-order-items');
            try {
                const items = JSON.parse(o.items || "[]");
                if(items.length === 0) list.innerHTML = "<p class='text-gray-400 italic text-center'>Data rincian tidak tersedia (Pesanan Lama).</p>";
                else list.innerHTML = items.map(i => `<div class="flex justify-between items-center border-b border-dashed pb-2"><div class="flex-grow"><p class="text-sm font-bold text-gray-800">${i.name}</p><p class="text-xs text-gray-500">${i.qty} x Rp ${Number(i.price).toLocaleString('id-ID')}</p></div><div class="font-bold text-brandRed text-sm">Rp ${(i.qty * i.price).toLocaleString('id-ID')}</div></div>`).join('');
            } catch(e) {
                list.innerHTML = "<p class='text-red-400 italic'>Gagal memuat rincian.</p>";
            }
            
            const m=document.getElementById('order-details-modal'), b=document.getElementById('order-details-box'); 
            m.classList.remove('hidden'); m.classList.add('flex'); setTimeout(()=>{m.classList.remove('opacity-0');b.classList.remove('scale-95');},10);
        }

        function closeOrderDetailsModal() { 
            const m=document.getElementById('order-details-modal'), b=document.getElementById('order-details-box'); 
            m.classList.add('opacity-0'); b.classList.add('scale-95'); setTimeout(()=>{m.classList.add('hidden');m.classList.remove('flex');},300); 
        }
